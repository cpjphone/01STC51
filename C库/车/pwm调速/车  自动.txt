#include "reg52.h"
#define uint unsigned int
#define uchar unsigned char
sbit ena=P1^4;
sbit enb=P1^5;
sbit in1=P1^0;
sbit in2=P1^1;
sbit in3=P1^2;
sbit in4=P1^3;
//sbit key=P2^7;
//sbit key2=P2^6;
void delay(uint z)
{
   uint a,b;
   for(a=z;a>0;a--)
      for(b=110;b>0;b--);
}
void qian()
{
  in1=1;
  in2=0;
  in3=1;
  in4=0;
}
void main()
 {
    uint m,n;
	m=1,n=1;
	while(1)
	{
   		if(n<1000)
   		{
			n++;
		 	if(m%2==0)
		 	{
			   	ena=1;
			   	enb=1;
			  	delay(4);
			  	m++;
		 	}
			else
		 	{
			  	ena=0;
			  	enb=0;
			  	qian();
			  	delay(6);
		  		m++;
		 	}
			delay(5);
		}
   		if(n>=1000)
   		{
		 	if(m%2==0)
	 		{
			   	ena=1;
			   	enb=1;
			   	delay(9);
			   	m++;
			}
			else
			{
			 	ena=0;
			  	enb=0;
			  	qian();
			  	delay(1);
			  	m++;
	 		}
			n++;
			delay(5);
			if(n==2000)
			n=0;
    	}
  	}
}
